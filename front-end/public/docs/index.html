어플리케이션 생성

br 함수를 사용하여 어플리케이션을 생성 및 초기화 합니다.
생성된 app는 즉시실행 함수로의 기능을 수행합니다.

const app= br('appName', app=> {
    ...
});

app(_=> {
    // 즉시 실행 함수 내부
});

app: BrApp 속성

app.debug: BrDebug

브라우저 콘솔 로그를 관리 합니다.

app.el: BrElement

HTML 엘리먼트 객체를 생성 합니다.

// html: html 검색
// async: 비동기 처리

html.set('alert', _=> {
    const result= html.query('#alert');
    // 비동기 호출을 대기 리스너
    async.supended(result, param=> {
        // 팝업 노출처리 공통화...
        console.log('----', param);
    });

    const button= result.query('button');
    button.addEventListener('click', e=> {
        // 비동기 귀결처리
        async.resolve(result, {a:1});
        // 비동기 거부처리
        async.reject(result, {a:1});
    });

    return result;
});
const pop= html.get('alert');

async.call(pop, {payload:{}})

        console.log('-종료결과-', rs);
    })
