Example
app(_=> {
   app.debug(true); // 로그 활성화
   app.debug.log('app.log', 1); // 콘솔 로그 출력
   app.debug.warn('app.warn', 1); // 경고 출력
   app.debug.error('app.error', 1); // 오류 출력
   app.debug.info('app.info', 1); // 강조(green)된 로그 출력
   app.debug.success('app.success', 1); // 강조(blue)된 로그 출력
   app.debug.danger('app.danger', 1); // 강조(red)된 로그 출력
});
