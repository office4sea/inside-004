Example
// <form>
//     <h1 data-vo="title">title<h1>
//     이름: <input data-vo="name" type="text">
//     <button data-vo="submit" type="button">확인</button>
//     <p data-vo="print">입력 결과</p>
// </form>

// dataVo(BrHtml.dataVo) 핸들 취득
const form= app.html.dataVo('form');

// 값 세팅
form.title.html= '<small>필수</small> 이름입력';
form.name.value= '';
form.print.text= '';

// 클릭 이벤트
form.button.event('click', e=> {
    if(!form.name.value) return alert('이름을 입력하세요.');

    // 값 취득
    console.log('VALUE:', form.name.value);
    console.log('HTML:', form.title.html);
    console.log('TEXT:', form.print.text);
});

Type:
  • object
Example
// form: BrHtmlElementItem 샘플 참고
// 노드 추가
const check= app.el.input({id:'check1', type:'checkbox'});
const label= app.el.label('체크박스', {for:'check1'});
const group= form.insert(app.el.div({class:'form-group'}, [
    check, label
]));

// 클랙스 추가
group.clss.add('a')  // 단건 추가
    .add(['b', 'c']) // 다건 추가

// 클릭 이벤트: BrHtmlElementItem.event
group.event('click', e=> {
    // 클랙스 추가
    if(check.checked) label.clss.add('active');
    else {
        // 클랙스 삭제
        label.clss.remove('active');
    }
    // 클랙스 확인
    console.log('체크 상태:', label.clss.has('active'));
});
